# Segnaposto per feature extractor
def identity(input):
    return input

# Data loader
def load_data(feature_extractor=identity, normalize=True):

    labels = []
    features = []

    for f in sorted(os.listdir('./recordings')):
        if f.endswith('.wav'):
            # Carica file ed estraine le features
            _, signal = wav.read('./recordings/' + f)
            cur_features = feature_extractor(signal)
            features.append(cur_features)

            # Categorie
            label = f.split('_')[0]
            labels.append(label)

    # X: features, y: labels
    X_train, X_test, y_train, y_test = train_test_split(features, labels, test_size=0.1, random_state=1)

    if normalize:
        eps = 0.001
        X_train = np.array(X_train)
        X_train_mean = X_train.mean(axis=0)
        X_train_std = X_train.std(axis=0)
        X_train = (X_train - X_train_mean + eps)/(X_train_std + eps)
        X_train = [row for row in X_train]

        X_test = [row for row in (np.array(X_test) - X_train_mean + eps)/(X_train_std + eps)]

    return X_train, X_test, y_train, y_test

def crop(input, size=100):
    # Elimina eventuali valori oltre il numero consentito
    output = input[0:min(size, input.shape[0])]
    # Aggiungi valori nulli per raggiungere la dimensione richiesta
    output = np.concatenate((output, np.zeros(size-output.shape[0])))
    return output

def sdev(input):
    return np.std(input, keepdims=True)

def avg(input):
    return np.mean(input, keepdims=True)

def aavg(input):
    return np.mean(np.abs(input), keepdims=True)

def duration(input):
    return input.shape

def energy(input):
    return np.sum((input*1.0)**2, keepdims=True)

3.2 Creazione della feature Zero-Crossing Rate (ZCR)¶

Per short-time average ZCR s'intende il numero di volte in cui il segnale attraversa lo zero in un tempo unitario. La funzione da implemenetare la seguente:

$$ ZCR = \frac{1}{2}\sum_{k=n}^{n+N+1}|sgn(s_w(k))-sgn(s_w(k-1))|$$

dove $sgn(\cdot)$ la funzione segno, $w$ la funzione finestra e $s_w(k)$ il segnale dopo il windowing processing.

In [6]:
def zcr(input):
    return np.sum(np.abs(np.diff(np.sign(input))), keepdims=True)*0.5

# funzione per la creazione della matrice di confusione
def confusion_map(real, pred):

    cm = confusion_matrix(real, pred)
    fig, ax = plt.subplots(figsize=(7, 5))
    im = ax.imshow(cm, cmap=plt.cm.Blues)

    ax.set_xticks(np.arange(10))
    ax.set_yticks(np.arange(10))

    ax.set_xticklabels(np.arange(10))
    ax.set_yticklabels(np.arange(10))

    for i in range(10):
        for j in range(10):
            text = ax.text(j, i, cm[i, j],
                           ha="center", va="center")

    ax.set_title("Matrice di confusione")
    fig.tight_layout()
    plt.show()

7.2 Considerazioni finali¶

Come prima cosa possibile osservare che la combinazione degli iper-parametri ottenuta dalla GridSearch differente per i due modelli. Infatti, il modello clf (il modello iniziale senza la feature zcr) ottiene la migliore perfomance con $C=3,500$ e $\gamma=0.5$. Il modello clf_zcr, invece, ottiene la migliore perormance con $C=2,500$ e $\gamma=0.1$. La funzione kernel migliore per entrambi i modelli risulta essere la RBF.
Infine, osservando le matrici di confusione e le relative perfomance sui dati di test possibile osservare che il modello in cui presente la feature zcr, ottiene risultati migliori: basandoci sulla misura macro avg, si ottiene un miglioramento di circa il 24%. Per cui possibile affermare che la features zcr, che pu riflettere la natura dello spettro del segnale in una certa misura, risulta avere un notevole impatto sulle performance del classificatore.
